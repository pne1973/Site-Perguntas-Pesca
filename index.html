async function consultarGemini() { const pergunta = document.getElementById('pergunta').value; const divResultado = document.getElementById('resultado'); if (!pergunta) return; divResultado.style.display = "block"; divResultado.innerText = "Consultando o Mestre..."; try { const response = await fetch('/api/chat', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ pergunta: pergunta }) }); const data = await response.json(); divResultado.innerText = data.text; } catch (error) { divResultado.innerText = "Erro ao carregar resposta."; } }
